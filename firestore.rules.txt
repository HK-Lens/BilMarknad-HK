rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    function signedIn() {
      return request.auth != null;
    }

    function isNonEmptyString(value) {
      return value is string && value.size() > 0;
    }

    function hasParticipant(data, uid) {
      return data.participants is list && uid in data.participants;
    }

    function isChatParticipant(data) {
      return signedIn()
        && (
          hasParticipant(data, request.auth.uid)
          || data.buyerId == request.auth.uid
          || data.sellerId == request.auth.uid
        );
    }

    function chatPath(chatId) {
      return /databases/$(database)/documents/chats/$(chatId);
    }

    function chatExists(chatId) {
      return exists(chatPath(chatId));
    }

    function currentChat(chatId) {
      return get(chatPath(chatId)).data;
    }

    function isParticipantOfExistingChat(chatId) {
      return signedIn()
        && chatExists(chatId)
        && isChatParticipant(currentChat(chatId));
    }

    function validChatCreate() {
      return signedIn()
        && request.resource.data.keys().hasOnly([
          'carId',
          'carName',
          'carTitle',
          'title',
          'participants',
          'buyerId',
          'sellerId',
          'buyerName',
          'buyerEmail',
          'sellerName',
          'sellerEmail',
          'participantNames',
          'participantEmails',
          'lastMessage',
          'lastSenderId',
          'unreadBy',
          'hiddenFor',
          'createdAt',
          'updatedAt'
        ])
        && isNonEmptyString(request.resource.data.buyerId)
        && isNonEmptyString(request.resource.data.sellerId)
        && isNonEmptyString(request.resource.data.carId)
        && request.resource.data.buyerId != request.resource.data.sellerId
        && request.resource.data.participants is list
        && request.resource.data.participants.size() >= 2
        && request.resource.data.participants.size() <= 10
        && request.resource.data.buyerId in request.resource.data.participants
        && request.resource.data.sellerId in request.resource.data.participants
        && request.auth.uid in request.resource.data.participants
        && request.resource.data.lastMessage is string
        && request.resource.data.lastMessage.size() <= 1500
        && request.resource.data.lastSenderId is string
        && request.resource.data.unreadBy is list
        && request.resource.data.hiddenFor is list;
    }

    function coreChatFieldsUnchanged() {
      return request.resource.data.buyerId == resource.data.buyerId
        && request.resource.data.sellerId == resource.data.sellerId
        && request.resource.data.carId == resource.data.carId;
    }

    function validParticipantsAfterUpdate() {
      return !request.resource.data.diff(resource.data).affectedKeys().hasAny(['participants'])
        || (
          request.resource.data.participants is list
          && resource.data.buyerId in request.resource.data.participants
          && resource.data.sellerId in request.resource.data.participants
          && request.auth.uid in request.resource.data.participants
        );
    }

    function validChatUpdate() {
      return isChatParticipant(resource.data)
        && coreChatFieldsUnchanged()
        && validParticipantsAfterUpdate()
        && request.resource.data.diff(resource.data).affectedKeys().hasOnly([
          'participants',
          'participantNames',
          'participantEmails',
          'buyerName',
          'buyerEmail',
          'sellerName',
          'sellerEmail',
          'carName',
          'carTitle',
          'title',
          'lastMessage',
          'lastSenderId',
          'unreadBy',
          'hiddenFor',
          'updatedAt'
        ]);
    }

    function validMessageCreate() {
      return signedIn()
        && request.resource.data.keys().hasOnly([
          'senderId',
          'senderEmail',
          'senderName',
          'text',
          'createdAt'
        ])
        && request.resource.data.senderId == request.auth.uid
        && isNonEmptyString(request.resource.data.text)
        && request.resource.data.text.size() <= 1500;
    }

    function validCarTermsAgreement() {
      return request.resource.data.termsAccepted == true
        && request.resource.data.publicationRightsConfirmed == true
        && request.resource.data.privacyPolicyAccepted == true
        && request.resource.data.termsVersion is string
        && request.resource.data.termsVersion.size() > 0
        && request.resource.data.termsAcceptedAt is timestamp;
    }

    match /users/{userId} {
      allow read, create, update, delete: if signedIn() && request.auth.uid == userId;
    }

    match /cars/{carId} {
      allow read: if true;
      allow create: if signedIn()
        && validCarTermsAgreement()
        && (
        request.resource.data.sellerId == request.auth.uid
        || request.resource.data.ownerId == request.auth.uid
        || request.resource.data.userId == request.auth.uid
        || request.resource.data.uid == request.auth.uid
        || request.resource.data.createdBy == request.auth.uid
        || request.resource.data.createdById == request.auth.uid
      );
      allow update, delete: if signedIn() && (
        resource.data.sellerId == request.auth.uid
        || resource.data.ownerId == request.auth.uid
        || resource.data.userId == request.auth.uid
        || resource.data.uid == request.auth.uid
        || resource.data.createdBy == request.auth.uid
        || resource.data.createdById == request.auth.uid
      );
    }

    match /carListings/{carId} { allow read: if true; }
    match /listings/{carId} { allow read: if true; }
    match /ads/{carId} { allow read: if true; }
    match /vehicles/{carId} { allow read: if true; }

    match /favorites/{favoriteId} {
      allow read: if signedIn() && resource.data.userId == request.auth.uid;
      allow create: if signedIn() && request.resource.data.userId == request.auth.uid;
      allow update: if signedIn()
        && resource.data.userId == request.auth.uid
        && request.resource.data.userId == request.auth.uid;
      allow delete: if signedIn() && resource.data.userId == request.auth.uid;
    }

    match /chats/{chatId} {
      // Important for new chats: getDoc() for a not-yet-created deterministic chat ID must not fail.
      allow get: if signedIn()
        && (!chatExists(chatId) || isChatParticipant(resource.data));
      allow list: if isChatParticipant(resource.data);
      allow create: if validChatCreate();
      allow update: if validChatUpdate();
      allow delete: if false;

      match /messages/{messageId} {
        allow read: if isParticipantOfExistingChat(chatId);
        allow create: if isParticipantOfExistingChat(chatId) && validMessageCreate();
        allow update: if false;
        allow delete: if isParticipantOfExistingChat(chatId)
          && resource.data.senderId == request.auth.uid;
      }
    }
  }
}
